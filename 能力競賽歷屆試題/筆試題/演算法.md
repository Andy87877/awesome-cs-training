```cpp

    if (maze[x][y] != ‘O’)
        return;
    maze[x][y] = ‘X’;
    for i = -1 to 1 do {
        for j = 0 to 1 do {
            dfs (maze, x+i, y+j);
        }
    }
}
```
```cpp
char maze[6][7] = {
    “XXXXXXX”,
    “XOXXXOX”,
    “XOXOOXX”,
    “XOOOXOX”,
    “XXOOOXX”,
    “XXXXXXX”
};
```

(B) 3
(C) 0
